% --- Definition/Theorem environments (Numbered with title format) ---
% Title format: "Number Type: Title" (e.g., "2.2.1 Definition: Real Vector Space")

% Definition environment (green)
\newtcbtheorem[number within=section]{definition}{Definition}{%
    my_thm_base, colframe=defgreen, colback=defgreen!5,
    separator sign={:},
    title={\thesection.\arabic{tcbcounter}\quad\tcbtitle}
}{def}

% Notation environment (green)
\newtcbtheorem[number within=section, use counter from=definition]{notation}{Notation}{%
    my_thm_base, colframe=defgreen, colback=defgreen!5,
    separator sign={:},
    title={\thesection.\arabic{tcbcounter}\quad\tcbtitle}
}{not}

% Example environment (horizontal lines style)
\newtcbtheorem[number within=section]{example}{}{%
    enhanced,
    breakable,
    frame hidden,             % Hide outer frame
    interior hidden,          % Hide background color
    sharp corners,            % Sharp corners
    boxrule=0pt,              % Base border thickness
    left=15pt, right=15pt,    % Left/right padding
    top=12pt, bottom=12pt,    % Internal vertical spacing
    toptitle=2pt,            % Spacing between title and top line
    before skip=12pt,        % Spacing before box
    after skip=12pt,         % Spacing after box
    % Top and bottom horizontal lines
    borderline north={1pt}{0pt}{black!50},
    borderline south={1pt}{0pt}{black},
    coltitle=black,           % Title color
    fonttitle=\sffamily\bfseries,      % Sans-serif bold title
    description font=\sffamily,
    attach title to upper,
    after title={\par\smallskip},
    separator sign none,
    title={\thesection.\arabic{tcbcounter}\quad Example: \tcbtitle},
}{ex}
