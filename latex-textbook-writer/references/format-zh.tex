% --- 定理环境统一样式 ---
\tcbset{
    my_thm_base/.style={
        enhanced,
        breakable,
        arc=6pt,                      % 外框圆角
        outer arc=6pt,
        boxrule=1pt,                  % 边框线宽
        width=\textwidth,
        center,
        left=10pt, right=10pt, top=15pt, bottom=8pt,
        before skip=18pt, after skip=15pt,
        fontupper=\rmfamily,          % 正文直体
        fonttitle=\bfseries\sffamily, % 标题粗体等线
        coltitle=white,
        separator sign none,
        titlerule=0pt,
        toptitle=1mm, bottomtitle=1mm,
    }
}

% --- 定义/定理环境（编号 + 标题格式）---
% 标题格式："编号 标题"（例如："2.2.1 抛硬币计数"）

% 定义环境（绿色）
\newtcbtheorem[number within=section]{definition}{}{%
    my_thm_base, colframe=defgreen, colback=defgreen!5,
    title={\thetcbcounter\quad \tcbtitle}
}{def}

% 记号环境（绿色）
\newtcbtheorem[number within=section, use counter from=definition]{notation}{}{%
    my_thm_base, colframe=defgreen, colback=defgreen!5,
    title={\thetcbcounter\quad \tcbtitle}
}{not}

% 定理环境（橙色）
\newtcbtheorem[number within=section]{theorem}{}{%
    my_thm_base, colframe=thmorange, colback=thmorange!5,
    title={\thetcbcounter\quad \tcbtitle}
}{thm}

% 引理环境（橙色）
\newtcbtheorem[number within=section, use counter from=theorem]{lemma}{}{%
    my_thm_base, colframe=thmorange, colback=thmorange!5,
    title={\thetcbcounter\quad \tcbtitle}
}{lem}

% 命题环境（橙色）
\newtcbtheorem[number within=section, use counter from=theorem]{proposition}{}{%
    my_thm_base, colframe=thmorange, colback=thmorange!5,
    title={\thetcbcounter\quad \tcbtitle}
}{prop}

% 例环境（蓝色）
\newtcbtheorem[number within=section]{example}{}{%
    my_thm_base, colframe=exblue, colback=exblue!5,
    title={\thetcbcounter\quad \tcbtitle}
}{ex}

% 练习环境（棕色）
\newtcbtheorem[number within=section]{exercise}{}{%
    my_thm_base, colframe=exercisehead, colback=exercisehead!5,
    title={\thetcbcounter\quad \tcbtitle}
}{exer}

% 注
\newtcolorbox{remarkbox}[1][]{
    enhanced, breakable, colback=remgray!20, colframe=remgray!20,
    boxrule=0pt, arc=4pt, left=6pt, right=6pt, top=6pt, bottom=6pt,
    before skip=10pt, after skip=10pt, fontupper=\sffamily\small,
    title={#1}
}
\newenvironment{remark}[1][]{\begin{remarkbox}[注：#1]}{\end{remarkbox}}
