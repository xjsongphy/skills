% 节标题格式（标题用等线+粗体；正文用宋体）
\titleformat{\chapter}[display]
{\sffamily\huge\bfseries\color{titleblue}}
{\chaptertitlename\ \thechapter}{20pt}{\Huge\bfseries}

% --- 定理环境统一样式 ---
\tcbset{
    my_thm_base/.style={
        enhanced,
        breakable,
        arc=6pt,
        outer arc=6pt,
        boxrule=1pt,
        width=\textwidth,
        center,
        left=10pt, right=10pt, top=15pt, bottom=8pt,
        before skip=22pt, after skip=20pt,
        fontupper=\rmfamily,          % 正文用衬线字体（宋体）
        fonttitle=\bfseries\sffamily, % 标题用等线+粗体
        coltitle=white,
        separator sign none,
        titlerule=0pt,
        toptitle=1mm, bottomtitle=1mm,
    }
}

% 定义环境（绿色）
\newtcbtheorem[number within=section]{definition}{定义}{%
    my_thm_base, colframe=defgreen, colback=defgreen!5,
    separator sign={：},
    title={\thesection.\arabic{tcbcounter}\quad\tcbtitle}
}{def}

% 记号环境（绿色）
\newtcbtheorem[number within=section, use counter from=definition]{notation}{记号}{%
    my_thm_base, colframe=defgreen, colback=defgreen!5,
    separator sign={：},
    title={\thesection.\arabic{tcbcounter}\quad\tcbtitle}
}{not}

% 定理环境（橙色）
\newtcbtheorem[number within=section]{theorem}{定理}{%
    my_thm_base, colframe=thmorange, colback=thmorange!5,
    separator sign={：},
    title={\thesection.\arabic{tcbcounter}\quad\tcbtitle}
}{thm}

% 引理环境（橙色）
\newtcbtheorem[number within=section, use counter from=theorem]{lemma}{引理}{%
    my_thm_base, colframe=thmorange, colback=thmorange!5,
    separator sign={：},
    title={\thesection.\arabic{tcbcounter}\quad\tcbtitle}
}{lem}

% 命题环境（橙色）
\newtcbtheorem[number within=section]{proposition}{命题}{%
    my_thm_base, colframe=thmorange, colback=thmorange!5,
    separator sign={：},
    title={\thesection.\arabic{tcbcounter}\quad\tcbtitle}
}{prop}

% 例环境（上下横线风格）
\newtcbtheorem[number within=section]{example}{}{%
    enhanced,
    breakable,
    frame hidden,             % 隐藏外框
    interior hidden,          % 隐藏背景色
    sharp corners,            % 直角
    boxrule=0pt,              % 基础边框厚度
    left=15pt, right=15pt,    % 左右留白
    top=12pt, bottom=12pt,    % 内部上下间距
    toptitle=2pt,            % 标题与顶部横线的间距
    before skip=12pt,        % 盒子与上方段落的间距
    after skip=12pt,         % 盒子与下方段落的间距
    % 设置上下两条横线（起始灰色，末尾黑色）
    borderline north={1pt}{0pt}{black!50},
    borderline south={1pt}{0pt}{black},
    coltitle=black,               % 标题颜色
    fonttitle=\bfseries\sffamily, % 标题用等线+粗体
    fontupper=\rmfamily,          % 正文用宋体
    attach title to upper,
    after title={\par\smallskip},
    separator sign none,
    title={\thesection.\arabic{tcbcounter}\quad 例：\tcbtitle},
}{ex}

% 练习环境（棕色）
\newtcbtheorem[number within=section]{exercise}{练习}{%
    my_thm_base, colframe=exercisehead, colback=exercisehead!5,
    separator sign={：},
    title={\thesection.\arabic{tcbcounter}\quad\tcbtitle}
}{exer}

% 注环境（深灰色）
\newtcbtheorem[number within=section]{remark}{注}{%
    my_thm_base, colframe=remgray, colback=remgray!5,
    separator sign={：},
    title={\thesection.\arabic{tcbcounter}\quad\tcbtitle}
}{rem}

% 证明（标题粗体，正文宋体）

}{%

}
